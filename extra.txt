        client.on("core_validateProject", (data) => {
            if (valid_auth) {
                let project = data.project
                console.log("checkng")
                console.log(db.isProject(project))
                if (db.isProject(project) && user_uuid != null) {
                    user_project = project
                    let pack = db.getPack(db.getProject(project).pack)
                    client.emit("core_validProject", { plugins: pack.plugins })
                } else {
                    client.emit("core_invalidProject")
                }
            }
        })
        let downloadAttempts = 0
        client.on("core_checkPlugins", (data) => {
            let plugins = data.plugins
            let mustDownload = []
            for (let plugin in plugins) {
                let hash = plugins[plugin]
                let stored = db.getPluginCompiledChecksum(plugin)
                console.log("Stored:" + stored)
                console.log("Hash: " + hash)
                if (stored != hash || stored == undefined) {
                    mustDownload.push(plugin)
                }
            }
            if (mustDownload.length > 0 && downloadAttempts < 4) {
                console.log("why")
                downloadAttempts++
                client.emit("core_downloadPlugins", { plugins: mustDownload })
            } else {
                client.emit("core_verifiedPlugins", { plugins: null })
            }
        })
